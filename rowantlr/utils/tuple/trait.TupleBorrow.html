Expand description

Get a shared reference for each element of a tuple.

Examples

let data: (String, Vec<&'static str>, u64) =
    ("answer".to_string(), vec!["life", "the universe", "everything"], 42);
let borrowed: (&String, &Vec<&'static str>, &u64) = data.tuple_borrow();
